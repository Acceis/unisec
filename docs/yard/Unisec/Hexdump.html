def self.utf16be(str)
  str.encode('UTF-16BE').to_hex.scan(/.{4}/).join(' ')
end

def display
  "UTF-8: #{@utf8}\n" \
    "UTF-16BE: #{@utf16be}\n" \
    "UTF-16LE: #{@utf16le}\n" \
    "UTF-32BE: #{@utf32be}\n" \
    "UTF-32LE: #{@utf32le}"
end
